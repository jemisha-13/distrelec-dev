UPDATE DISTMANUFACTURER SET P_GLOBALSUPPORTREPAIRINFO_CLOB=P_GLOBALSUPPORTREPAIRINFO;

ALTER TABLE DISTMANUFACTURER DROP COLUMN P_GLOBALSUPPORTREPAIRINFO;
